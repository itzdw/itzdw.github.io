vue-v-directives

vue自定义指令扩展

快速上手 →

# 安装方便

npm i -S vue-v-directives
// or
yarn add vue-v-directives

# 使用简单

import Vue from 'Vue'
import directives from 'vue-v-directives'
Vue.use(directives)

# 按需加载，防止冲突

import Vue from 'Vue'
import directives from 'vue-v-directives'
directives.use(['clickOutside', 'watermarker', 'copy', 'debounce', 'limitText'])
Vue.use(directives)
